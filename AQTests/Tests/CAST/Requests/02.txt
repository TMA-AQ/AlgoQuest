SELECT 
CASE CAST(SUBSTRING (TD_HEBDO_CALENDRIER.seq_sem_court, 5, 2) AS INTEGER)
    WHEN 1
    THEN TD_HEBDO_CALENDRIER.SEQ_SEM_COURT
    WHEN 2
    THEN TD_HEBDO_CALENDRIER.SEQ_SEM_COURT
	ELSE 'NO MATCH'
END
FROM  TD_HEBDO_CALENDRIER ,  TW_FLUX_QUOTIDIEN_HEBDO_DIAG 
WHERE TD_HEBDO_CALENDRIER.SEQ_SEM_COURT = TW_FLUX_QUOTIDIEN_HEBDO_DIAG.SEQ_SEM_COURT
AND  TD_HEBDO_CALENDRIER.seq_sem_court like '2010%';
