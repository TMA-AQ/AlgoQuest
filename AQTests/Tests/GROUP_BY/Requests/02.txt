SELECT TW_FLUX_QUOTIDIEN_HEBDO_DIAG.SEQ_PRODUIT
FROM TW_FLUX_QUOTIDIEN_HEBDO_DIAG, TD_HEBDO_CALENDRIER
WHERE TW_FLUX_QUOTIDIEN_HEBDO_DIAG.SEQ_SEM_COURT= TD_HEBDO_CALENDRIER.SEQ_SEM_COURT
AND TW_FLUX_QUOTIDIEN_HEBDO_DIAG.SEQ_PRODUIT='AD4208'
GROUP BY TW_FLUX_QUOTIDIEN_HEBDO_DIAG.seq_caf_histo,
TW_FLUX_QUOTIDIEN_HEBDO_DIAG.seq_produit,
TW_FLUX_QUOTIDIEN_HEBDO_DIAG.seq_indicateur
order by TW_FLUX_QUOTIDIEN_HEBDO_DIAG.SEQ_PRODUIT;
