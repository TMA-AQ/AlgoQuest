SELECT TAUX.val_3, count(*)
FROM   (SELECT table_1.val_3,
               tableAux.champ_1
        FROM   (SELECT table_6.champ_1
                FROM   table_6,
                       table_7
                WHERE  table_6.val_1 = table_7.val_1) tableAux,
               table_1
        WHERE  tableAux.champ_1 = table_1.champ_1
               AND table_1.champ_1 < 'abcabcghi') TAUX
WHERE  TAUX.champ_1 = 'abcabcdef'
group by TAUX.val_3
order by TAUX.val_3;
